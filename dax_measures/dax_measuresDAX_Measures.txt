3. Top Day = var tbl= SUMMARIZE('Balaji Fast Food Sales (1)','Balaji Fast Food Sales (1)'[date],"amt",[Total Amount]) RETURN maxx (TOPN(1,tbl,[amt],DESC), 'Balaji Fast Food Sales (1)'[date])

4. Total Amount = SUM('Balaji Fast Food Sales (1)'[transaction_amount])

5. Total price = SUM('Balaji Fast Food Sales (1)'[item_price])

6. Total Quantity = SUM('Balaji Fast Food Sales (1)'[quantity])

7. Monthly Growth % = VAR PrevMonth = CALCULATE([Total Amount], DATEADD('Balaji Fast Food Sales (1)'[date], -1, MONTH)) RETURN DIVIDE([Total Amount] - PrevMonth, PrevMonth,0)

9. Average item Price = AVERAGE('Balaji Fast Food Sales (1)'[item_price])

10. Amount_by_payment_type = CALCULATE([Total Amount], VALUES('Balaji Fast Food Sales (1)'[transaction_type]))
